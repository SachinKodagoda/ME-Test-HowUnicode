JavaScript (ES6 and Beyond) Cheat Sheet
The Ternary Operator
/* ---- Longhand ---- */
const x = 20 ;
let answer ;
if (x > 10) {
    answer = 'greater than 10' ;
} else {
    answer = 'less than 10' ;
}

/* ---- Shorthand ---- */
const answer = x > 10 ? 'greater than 10' : 'less than 10' ;
Short-circuit Evaluation
/* ---- Longhand ---- */
if ( v1 !== null || v1 !== undefined || v1 !== '') {
    v2 = v1;
} else {
    v2 = 'new'
}

/* ---- Shorthand ---- */
let v2 = v1 || 'new';
Declaring Variables
/* ---- Longhand ---- */
let x;
let y;
let z = 3 ;

/* ---- Shorthand ---- */
let x, y, z = 3 ;
