JavaScript (ES6 and Beyond) Cheat Sheet
EcmaScript (ES) is the standardised scripting language forJavaScript (JS). The current ES version supported in modern browsers is ES5. However, EcmaScript 6th edition (ES6 or ES2015) tackles a lot of the limitations of the core language, making it easier for developers to code.
Declaring Variables
Longhand ( ES5 )
var x ;
var y ;
var z = 3 ;
Shorthand ( ES5 )
var x, y, z = 3 ;
NOTE: In ES6, let or const is used instead of var keyword. const is a signal that the variable wonâ€™t be reassigned. let is a signal that the variable may be reassigned.
The Ternary Operator
Longhand ( ES5 )
var x = 20 ;
var answer ;
if ( x > 10 ) {
    answer = 'greater than 10' ;
} else {
    answer = 'less than 10' ;
}
console.log ( answer ) ;
Shorthand ( ES5 )
var x = 20 ;
var answer ;
answer = x > 10 ? 'greater than 10' : 'less than 10' ;
console.log ( answer ) ;
// console: greater than 10
NOTE:
if x is greater than 10 then pass x value, else pass 10. for that you can use following example
   answer = x > 10 ? x : 10 ;
Short-circuit Evaluation
Longhand ( ES5 )
var v1, v2;
if ( v1 === null || v1 === undefined || v1 === '' ) {
    v2 = 'no value' ;
} else {
    v2 = v1 ;
}
console.log ( v2 ) ;
Shorthand ( ES5 )
var v1 ;
var v2 = v1 || 'no value' ;
console.log ( v2 ) ;
// console: no value
If Presence
Longhand ( ES5 )
var v1 = true ;
if ( v1 === true ) {
   console.log ( "Yes" ) ;
} else {
   console.log ( "No" ) ;
}
Shorthand ( ES5 )
var v1 = true ;
if ( v1 ) {
   console.log ( "Yes" ) ;
} else {
   console.log ( "No" ) ;
}
// console: Yes
Longhand ( ES5 )
var v2 = false ;
if ( v2 !== true ) {
   console.log ( "Hi" ) ;
Shorthand ( ES5 )
var v2 = false ;
if ( !v2 ) {
   console.log ( "Hi" ) ;
}
// console: Hi
NOTE: above examples are correct only if v1 and v2 truthy values ( v1, v2 should be true or false only).
JavaScript for Loop
Longhand ( ES5 )
var arr = [ 1, 2, 3 ] ;
for (var i = 0 ; i < arr.length ; i++) {
   console.log ( i ) ;
}
Shorthand ( ES6 )
const arr = [ 1, 2, 3 ] ;
for ( let index of arr ) {
   console.log ( index );
}
// console:
     1
     2
     3

function logArrayElements ( element, index, array ) {

[2, 5, 9].forEach ( logArrayElements ) ;

Default Parameter Values
Longhand ( ES5 )
function volume ( l, w, h ) {
   if ( w === undefined )
     w = 3 ;
   if ( h === undefined )
     h = 4 ;
   return l * w * h ;
}
console.log ( volume ( 2 ) ) ;
Shorthand ( ES6 )
volume = ( l, w = 3, h = 4 ) => ( l * w * h ) ;
console.log ( volume ( 2 ) ) ;
// console: 24

// console [ 1, 2, 3, 4 ]

function findDog ( name ) {

Destructuring Assignment
Longhand
function validate(values) {
   if ( !values.first )
     return false;
   if( !values.last )
     return false;
   return true;
}
console.log ( validate ( { first : 'Bruce', last : 'Wayne' } ) ) ;
// true
Shorthand
// object validation rules
const schema = {
   first: {
     required : true
   },
   last: {
     required : true
   }
}
// universal validation function
const validate = ( schema, values ) => {
   for ( field in schema ) {
     if ( schema[field].required ) {
         if ( !values[field] ) {
             return false;
           }
       }
    }
   return true;
 }
console.log ( validate ( schema, { first : 'Bruce' } ) );
console.log ( validate ( schema, { first : 'Bruce' , last : 'Wayne' } ) );
